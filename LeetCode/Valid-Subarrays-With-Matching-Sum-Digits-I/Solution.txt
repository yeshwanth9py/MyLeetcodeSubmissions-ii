1class Solution {
2public:
3    int countValidSubarrays(vector<int>& nums, int x) {
4        int n = nums.size();
5        long long ans = 0;
6        //3123
7        for(int i=0; i<n; i++){
8            long long cs = 0;
9            for(int j=i; j<n; j++){
10                cs += nums[j];
11                int ld = (cs%10);  
12                int noz = log10(cs);
13                long long bas = pow(10, noz);
14                int fd = cs/bas;
15                if(ld == x && fd == x){
16                    ans++;
17                }
18            }
19        }
20
21        return ans;
22    }
